// LanguageSelector component: minimal, accessible, and client-side only.
// Place this at the top of each chapter .mdx (or import it from a shared component file).

export function LanguageSelector({ defaultLang = 'en' }) {
  const [lang, setLang] = React.useState(defaultLang)

  React.useEffect(() => {
    // keep selection in localStorage so user choice persists across pages
    try { localStorage.setItem('book_lang', lang) } catch (e) {}
  }, [lang])

  return (
    <div className="language-selector" aria-label="Choose language" style={{ marginBottom: '1rem' }}>
      <div style={{ display: 'flex', gap: '0.5rem', flexWrap: 'wrap' }}>
        <button onClick={() => setLang('en')} aria-pressed={lang === 'en'}>English</button>
        <button onClick={() => setLang('ur')} aria-pressed={lang === 'ur'}>اردو</button>
        <button onClick={() => setLang('ru')} aria-pressed={lang === 'ru'}>Roman Urdu</button>
        <button onClick={() => setLang('ar')} aria-pressed={lang === 'ar'}>العربية</button>
        <button onClick={() => setLang('de')} aria-pressed={lang === 'de'}>Deutsch</button>
      </div>

      <style>{`
        .language-selector button { padding: 0.45rem 0.7rem; border-radius: 6px; border: 1px solid rgba(0,0,0,0.08); background: transparent; cursor: pointer; }
        .language-selector button[aria-pressed="true"] { box-shadow: 0 0 0 3px rgba(0,0,0,0.04) inset; font-weight: 600; }
      `}</style>

      {/* Hidden inputs to allow CSS-only targeting if desired */}
      <input type="hidden" id="selected-lang" value={lang} />

      {/* Render translations container statefully */}
      <div>
        {/* The translation blocks below are always present in the .mdx (for RAG / offline retrieval),
            but they include a data-language attribute and are hidden unless selected. */}
        <div data-language="en" style={{ display: lang === 'en' ? 'block' : 'none' }}>
          {/* English content will appear here (can duplicate main content or provide summary). */}
        </div>
        <div data-language="ur" style={{ display: lang === 'ur' ? 'block' : 'none' }}>
          {/* Urdu translation block */}
        </div>
        <div data-language="ru" style={{ display: lang === 'ru' ? 'block' : 'none' }}>
          {/* Roman Urdu translation block */}
        </div>
        <div data-language="ar" style={{ display: lang === 'ar' ? 'block' : 'none' }}>
          {/* Arabic translation block */}
        </div>
        <div data-language="de" style={{ display: lang === 'de' ? 'block' : 'none' }}>
          {/* German translation block */}
        </div>
      </div>
    </div>
  )
}

      ## From LLMs to VLMs to VLA

      Vision-Language-Action (VLA) systems represent the cutting edge of embodied AI, bridging the gap between perception, reasoning, and physical action. These systems combine:

      - **Vision**: Understanding the visual environment
      - **Language**: Interpreting human instructions and context
      - **Action**: Executing physical tasks in the real world

      ### Evolution of AI Systems

      1. **Large Language Models (LLMs)**: Text-based reasoning and generation
      2. **Vision-Language Models (VLMs)**: Visual understanding with language context
      3. **Vision-Language-Action (VLA) Systems**: Physical interaction with the world

      ## Connecting LLMs to Robots

      Integrating large language models with robotic systems requires careful consideration of:

      - **Prompt Engineering**: Crafting effective instructions for robot tasks
      - **Action Space Mapping**: Converting language commands to robot actions
      - **Perception Integration**: Using sensor data to inform language models
      - **Safety Constraints**: Ensuring safe execution of language-directed actions

      ### Architecture Pattern

      ```
      Human Command → LLM → Task Planner → Action Executor → Robot
      ```

      ## Voice-to-Action using Whisper

      Voice interfaces enable natural human-robot interaction using speech recognition:

      - **Whisper Integration**: Real-time speech-to-text conversion
      - **Intent Recognition**: Understanding user intentions from spoken commands
      - **Response Generation**: Providing audible feedback to users
      - **Context Awareness**: Maintaining conversation context across multiple interactions

      ## Real-World Task Execution

      VLA systems excel at executing complex real-world tasks:

      - **Object Manipulation**: Picking, placing, and manipulating objects
      - **Navigation Tasks**: Moving through environments based on language commands
      - **Multi-step Operations**: Completing complex sequences of actions
      - **Human Collaboration**: Working alongside humans in shared spaces

      ## Safety & Constraints

      Safety is paramount in VLA systems:

      - **Physical Safety**: Preventing harm to humans and environment
      - **Logical Constraints**: Ensuring actions make sense in context
      - **Ethical Guidelines**: Following ethical principles in action selection
      - **Fail-Safe Mechanisms**: Graceful degradation when uncertain
